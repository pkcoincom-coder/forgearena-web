// ===== COC ID =====
function formatCocId(el) {
  let v = el.value.toUpperCase().replace(/[^A-Z0-9#]/g,'');
  if (v && !v.startsWith('#')) v = '#' + v;
  el.value = v;
  // Update profile link
  const tag = v.replace('#','');
  if (tag.length > 3) {
    document.getElementById('cocProfileLink').href =
      `https://link.clashofclans.com/th?action=OpenPlayerProfile&tag=${tag}`;
  }
}

  window.open(url, '_blank');
  showToast('🔍','Opening CoC Profile','Verify your tag in the game');
}

let lang = 'en';
function setLang(l) {
  lang = l;
  const t = T[l] || T.en;
  document.querySelectorAll('[data-i18n]').forEach(el => {
    const k = el.getAttribute('data-i18n');
    if (t[k] !== undefined) el.innerHTML = t[k];
  });
  document.querySelectorAll('.lbtn').forEach(b => b.classList.remove('on'));
  const ab = document.getElementById('btn-' + l);
  if (ab) ab.classList.add('on');
  const fl = document.getElementById('fLang');
  if (fl) for (let o of fl.options) if (o.value === l) { fl.value = l; break; }
  document.documentElement.lang = l;
  // Update RTL for Arabic
  document.body.style.direction = l === 'ar' ? 'rtl' : 'ltr';
}

function setLangFromSelect() { const v = document.getElementById('fLang').value; if (v) setLang(v); }
function autoLang() {
  const sel = document.getElementById('fCountry');
  const opt = sel.options[sel.selectedIndex];
  const dl = opt ? opt.getAttribute('data-lang') : null;
  if (dl && T[dl]) setLang(dl);
}

async function handleSubmit() {
  const name = document.getElementById('fName').value.trim();
  const email = document.getElementById('fEmail').value.trim();
  const country = document.getElementById('fCountry').value;
  const game = document.getElementById('fGame').value;
  const errEl = document.getElementById('formError');

  if (!name || !email || !country || !game || !selectedRole) {
    errEl.classList.add('show');
    setTimeout(() => errEl.classList.remove('show'), 3000);
    return;
  }

  const btn = document.getElementById('submitBtn');
  btn.innerHTML = '<span class="loading-spinner"></span> Saving...';
  btn.style.opacity = '.7';

  const channels = [...document.querySelectorAll('.chip.on')].map(c => c.textContent.trim()).join(', ');
  const memberId = 'FA-' + Date.now().toString(36).toUpperCase();

  const memberData = {
    name,
    email,
    role: selectedRole,
    country,
    language: document.getElementById('fLang').value || lang,
    coc_id: document.getElementById('fCocId').value || null,
    clan_name: document.getElementById('fClan')?.value || null,
    clan_size: document.getElementById('fClanSize')?.value || null,
    game,
    contact_channels: channels,
    contact_handle: document.getElementById('fHandle').value || null,
    member_id: memberId,
    registered_at: new Date().toISOString()
  };

  const result = await saveMember(memberData);

  document.getElementById('formArea').style.display = 'none';
  document.getElementById('successBox').style.display = 'block';
  document.getElementById('successMemberId').textContent = 'Member ID: ' + memberId;

  // Update stats
  total++;
  newToday++;
  document.getElementById('statTotal').textContent = fmtNum(total);
  document.getElementById('statNew').textContent = '+' + fmtNum(newToday);

  showToast('🎉', name + ' joined!', 'Member ID: ' + memberId);
}

function animatePrize() {
  let start=100000,end=prize,dur=2000,startTime=null;
  function step(ts){
    if(!startTime)startTime=ts;
    let p=Math.min((ts-startTime)/dur,1);
    let e=1-Math.pow(1-p,3);
    document.getElementById('prizeAmount').textContent=fmtNum(Math.floor(start+(end-start)*e));
    if(p<1)requestAnimationFrame(step);
  }
  requestAnimationFrame(step);
}

function drawThumb(id, c1, c2) {
  const c = document.getElementById(id);
  if (!c) return;
  const ctx = c.getContext('2d');
  const w = c.width, h = c.height;
  let t = 0;
  function draw() {
    ctx.fillStyle='#080810';ctx.fillRect(0,0,w,h);
    ctx.strokeStyle='rgba(255,61,0,0.08)';ctx.lineWidth=1;
    for(let x=0;x<w;x+=24){ctx.beginPath();ctx.moveTo(x,0);ctx.lineTo(x,h);ctx.stroke();}
    for(let y=0;y<h;y+=24){ctx.beginPath();ctx.moveTo(0,y);ctx.lineTo(w,y);ctx.stroke();}
    for(let i=0;i<4;i++){
      const y=h-36+(i%2)*16;const maxW=w-40;
      const bw=(Math.sin(t*0.05+i*1.2)+1)/2*maxW;
      const g=ctx.createLinearGradient(20,0,20+bw,0);g.addColorStop(0,c1);g.addColorStop(1,c2);
      ctx.fillStyle='rgba(0,0,0,0.4)';ctx.fillRect(20,y,maxW,5);
      ctx.fillStyle=g;ctx.fillRect(20,y,bw,5);
    }
    ctx.beginPath();ctx.strokeStyle=c1;ctx.lineWidth=1.5;ctx.globalAlpha=0.4;
    for(let x=0;x<w;x++){const y2=h/2+Math.sin((x+t)*0.08)*12+Math.sin((x+t)*0.2)*4;if(x===0)ctx.moveTo(x,y2);else ctx.lineTo(x,y2);}
    ctx.stroke();ctx.globalAlpha=1;
    for(let s=0;s<15;s++){const sx=(s*137.5+t*0.3)%w;const sy=(s*73.1+t*0.1)%h;ctx.fillStyle='rgba(255,255,255,0.3)';ctx.beginPath();ctx.arc(sx,sy,Math.abs(Math.sin(t*0.02+s))*1.5,0,Math.PI*2);ctx.fill();}
    ctx.font='bold 10px sans-serif';ctx.fillStyle=c1;ctx.globalAlpha=0.7;
    ctx.fillText('TH17 ████ '+Math.floor(Math.abs(Math.sin(t*0.03))*100)+'%',20,22);
    ctx.fillText('★★★',20,38);ctx.globalAlpha=1;
    t++;requestAnimationFrame(draw);
  }
  draw();
}
